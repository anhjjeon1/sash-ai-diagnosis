import qrcode import matplotlib.pyplot as plt # 선생님의 최종 앱 주소 (v2 최종완성) url = "https://anhjjeon1.github.io/sash-ai-diagnosis/index.html" # QR코드 생성 qr = qrcode.QRCode( version=1, error_correction=qrcode.constants.ERROR_CORRECT_H, # 오류 복원력 높음 box_size=10, border=4, ) qr.add_data(url) qr.make(fit=True) img = qr.make_image(fill_color="black", back_color="white") # 시각화 plt.figure(figsize=(5, 5)) plt.imshow(img, cmap='gray') plt.axis('off') plt.show()
